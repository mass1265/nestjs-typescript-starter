  </div>

  <script>
    let progressValue = 0;
    const progress = document.getElementById('myProgress');

    // Define 10 colors (for each 10% step)
    const colors = [
      "#f44336", // 0-10% red
      "#e91e63", // 10-20%
      "#9c27b0", // 20-30%
      "#673ab7", // 30-40%
      "#3f51b5", // 40-50%
      "#2196f3", // 50-60%
      "#03a9f4", // 60-70%
      "#4caf50", // 70-80%
      "#8bc34a", // 80-90%
      "#ff9800"  // 90-100%
    ];

    // Store the interval ID
    const intervalId = setInterval(() => {
      progressValue += 10;
      progress.value = progressValue;

      // Pick color based on step
      const colorIndex = Math.floor(progressValue / 10) - 1;
      const color = colors[colorIndex] || "#4caf50";

      // Apply color dynamically
      progress.style.setProperty('--progress-color', color);

      // Chrome/Safari
      progress.style.setProperty('accent-color', color);

      // Firefox (fallback)
      const style = document.createElement('style');
      style.textContent = `
        progress[value]::-webkit-progress-value { background-color: ${color}; }
        progress[value]::-moz-progress-bar { background-color: ${color}; }
      `;
      document.head.appendChild(style);

      // Check if progress is complete
      if (progressValue >= 100) {
        progressValue = 100; // Ensure it doesn't exceed 100
        progress.value = 100;

        // Apply red color when complete
        const completionColor = "#ff0000"; // Bright red
        progress.style.setProperty('--progress-color', completionColor);
        progress.style.setProperty('accent-color', completionColor);

        // Firefox (fallback)
        const style = document.createElement('style');
        style.textContent = `
          progress[value]::-webkit-progress-value { background-color: ${completionColor}; }
          progress[value]::-moz-progress-bar { background-color: ${completionColor}; }
        `;
        document.head.appendChild(style);

        console.log('Progress complete');
        clearInterval(intervalId); // Stop the interval
      }
    }, 1000);

    // Optional: Add a button to manually stop the progress
    const stopButton = document.createElement('button');
    stopButton.textContent = 'Stop Progress';
    stopButton.style.marginTop = '20px';
    stopButton.onclick = () => {
      clearInterval(intervalId);
      console.log('Progress stopped manually');
    };
    document.body.appendChild(stopButton);
  </script>
</body>

    function renderCheckboxList() {
      const container = document.getElementById('result');
      container.innerHTML = frameworks.map(item => `
            <div class="checkbox-item">
                <input type="checkbox" 
                    id="${item.id}" 
                    name="${item.id}"
                    ${selectedItems.some(selected => selected.id === item.id) ? 'checked' : ''}
                    onChange="handleCheckboxChange(${item.id})"
                >
                <label for="${item.id}">${item.name}</label>
            </div>
        `).join('');
    }

    function renderSelectedList() {
      const listContainer = document.getElementById('list');
      listContainer.innerHTML = selectedItems.map(item => `
            <div class="selected-item">
                <span>${item.name}</span>
                <button class="remove-btn" onClick="removeItem(${item.id})">×</button>
            </div>
        `).join('');
    }

    // Event Handlers
    function handleCheckboxChange(id) {
      const checkbox = document.getElementById(id);
      if (checkbox.checked) {
        const itemToAdd = frameworks.find(item => item.id === id);
        selectedItems.push(itemToAdd);
      } else {
        selectedItems = selectedItems.filter(item => item.id !== id);
      }
      renderSelectedList();
    }

    function removeItem(id) {
      selectedItems = selectedItems.filter(item => item.id !== id);
      const checkbox = document.getElementById(id);
      if (checkbox) {
        checkbox.checked = false;
      }
      renderSelectedList();
    }

    // Initial render
    renderCheckboxList();

function findTopFour(arr) {
  let first = -Infinity;
  let second = -Infinity;
 

  for (let i = 0; i < arr.length; i++) {
    let num = arr[i];

    // Skip duplicates
    if (num === first || num === second ) {
      continue;
    }

    if (num > first) {
      second = first;
      first = num;
    } else if (num > second) {
      second = num;
    }  
  }

  return {
    first: first !== -Infinity ? first : null,
    second: second !== -Infinity ? second : null,
 
  };
}

let & const hoisting:
Hoisted: allocation of memory before it's initialization.

❓Do let and const variables get hoisted?

Ans: Yes, they are hoisted, but they are in Temporal Dead Zone.

Temporal Dead Zone (TDZ):
T.D.Z is a time between when a let variable was hoisted and until it is initialized with some value.

(Allocated memory in different space without getting attached to the Global object)
Note:
In JS, variables declared using var keyword are ⬅️ get attached to the Global object (window in browsers or this in global scope).

🔹 Are let and const hoisted?
Yes, both let and const are hoisted —
BUT unlike var, they are placed in a special state called the Temporal Dead Zone (TDZ).

🔸 What is Hoisting?
Hoisting refers to JavaScript's behavior of moving variable and function declarations to the top of their containing scope during compilation.

function findDuplicateCharacters(str) {
    const charCount = {};
    const duplicates = [];

    for (const char of str) {
        charCount[char] = (charCount[char] || 0) + 1;
    }

    for (const char in charCount) {
        if (charCount[char] > 1) {
            duplicates.push(char);
        }
    }

    return duplicates;
}

==========
var arr = [9, 4, 1, 7, 3, 9,10,15,9,8];

function findTopFour(arr) {
  let first = -Infinity;
  let second = -Infinity;


  for (let i = 0; i < arr.length; i++) {
    let num = arr[i];

    // Skip duplicates
    if (num === first || num === second ) {
      continue;
    }

    if (num > first) {
      second = first;
      first = num;
    } else if (num > second) {
      second = num;
    }
  }

  return {
    first: first !== -Infinity ? first : null,
    second: second !== -Infinity ? second : null,

  };
}

console.log(findTopFour(arr));
